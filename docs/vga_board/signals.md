
## Horizontal signals

### Common

A = (!1 && !4) && (64 && 128)

B = 2 && !8

### Trigger values

200 = A !&& ((!2 && 8) && (!16 && !32))

210 = A !&& (B && (16 && !32))

242 = A !&& (B && (16 && 32))

264 = 8 !&& 256



## Vertical signals

### Common

A = (!2 && 16) && (64 && 512)

B = 8 && !32

### Trigger values

600 = A !&& (B && (!1 && !4))

601 = A !&& (B && (1 && !4))

605 = A !&& (B && (1 && 4))

628 = A !&& (4 && 32)
